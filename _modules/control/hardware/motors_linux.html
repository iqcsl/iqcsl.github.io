[docs]def initMotor(sn) -> GenericStage:
    """
    Search for motor with given serial number and connect to it

    :type sn: str or int
    :param sn: serial number
    :returns: thorpy stage object
    """
    devices = list_devices()
    dev = None
    for d in devices:
        if d['serial_number'] == str(sn):
            dev = d; break
    if dev == None:
        puterr('Could not initialize motor with serial number ' + str(sn))
        return None

    port_name = dev['device']
    port = Port.create(port_name, str(sn))
    stages = port.get_stages()
    assert len(stages) == 1
    for stage in stages.values():
        return stage

    """
    Thorlabs DDR25 rotation stage

    Typically connected to a controller (e.x. KBD101)
    
    Example::

        from control.hardware.motors import DDR25
        import time
        
        motor = DDR25(12345678) # connect to motor/controller with serial number 12345678
        motor.home() # home motor

        motor.max_velocity = 100 # set maximum velocity to 100 deg/sec
        motor.acceleration = 150 # set acceleration to 150 deg/sec^2
        motor.goto(123) # go to 123 degrees

        print(motor.position) # should print something close to 123

        motor.jog_max_velocity = 200 # set maximum jog velocity to 200 deg/sec
        motor.jog_acceleration = 500 # set jog acceleration to 500 deg/sec
        motor.jog(2) # jog backwards
        time.sleep(1) # wait for it to get up to speed
        print(motor.velocity) # should print something close to -200
        print(motor.position)

        motor.stop() # stop in a profiled manner

    See below for a comprehensive description of all functionalities
    """

    def __init__(self, sn):
        """
        :type sn: int or str
        :param sn: serial number (usually found on the controller, not the rotation stage itself)

        :raises Exception: if fails to connect
        """

        self.stage = initMotor(sn)
        """The underlying thorpy stage"""

        if self.stage == None:
            raise Exception # should be caught by caller
        self.serial_number = sn
        """Serial number"""

        # model-dependent conversion factor
        self.stage._T_value = 102.4e-6 # couldn't find how this was set in thorpy so I just manually set it here

    # TODO implement close() like on windows side

    @property
    def is_connected(self) -> bool:
        """
        (Read-only) Whether or not the motor is connected

        For example::

            motor = DDR25() # successfully connect device
            print(motor.is_connected) # should print True
            # * unplug device *
            print(motor.is_connected) # should print False
        """
        # not sure if there's a better way to do this
        # right now we just attempty an identify() to see if it fails
        # because that worked during testing
        # passive things like checking position don't seem to work
        try:
            self.identify()
            return True
        except Exception as e:
            log('DDR25.is_connected() encountered the following exception:')
            log(repr(e))
            log(traceback.format_exc())
            return False

[docs]    def home(self, blocking:bool=True, timeout:float=5) -> bool:
        """
        Tell the motor to go home (0 degrees)

        This should usually be performed after the motor is connected,
        in order to calibrate it.

        For example::

            motor = DDR25(12345678)

            print(motor.position) # unreliable
            motor.goto(45) # unreliable

            motor.home() # go home

            print(motor.position) # reliable
            motor.goto(45) # reliable
        
        :param blocking: if True, wait for the motor to finish homing before continuing execution
        :param timeout: how many seconds to wait for home to complete
        :raises HardwareException: if timeout exceeded
        :returns: boolean indicating success
        """

        puts('Homing motor...')
        if self.in_motion:
            self.stop()
        self.stage.home_completed = False

        self.stage._port.send_message(MGMSG_MOT_MOVE_HOME(chan_ident = self.stage._chan_ident))
        if blocking:
            t = 0
            while not self.stage.home_completed:
                time.sleep(0.1)
                t += 0.1
                if t >= timeout:
                    puterr('Exceeded time limit waiting for motor to home')
                    raise HardwareException(str(self.serial_number) + ' home')
                    return False

            puts('Homed motor')
        return True



[docs]    def goto(self, pos:float, blocking:bool=True, timeout:float=5) -> bool:
        """
        Tell the motor to move to a given absolute position

        :param pos: position to move to, in degrees
        :param blocking: if True, wait for the motor to complete the move before continuing execution
        :param timeout: how many seconds to wait for move to complete
        :raises HardwareException: if timeout exceeded
        :returns: boolean indicating success
        """

        if self.in_motion:
            self.stop()
        self.stage.move_completed = False

        puts(f'Moving motor to {pos}...')
        scaled = pos/360
        self.stage.position = scaled
        if blocking:
            t = 0
            while not self.stage.move_completed:
                time.sleep(0.1)
                t += 0.1
                if t >= timeout:
                    puterr(f'Exceeded time limit waiting for motor to goto position {pos}')
                    raise HardwareException(str(self.serial_number) + ' goto')
                    return False
            puts(f'Moved motor to {pos}')
        return True



[docs]    def moveby(self, amt, blocking:bool=True, timeout:float=5) -> bool:
        """
        Tell the motor to move relatively by a given angle
        
        Example::
            
            motor = DDR25(12345678)
            motor.home()
            motor.moveby(720) # make 2 revolutions in the forward direction
            motor.moveby(-720) # make 2 revolutions in the backward direction

        :param amt: angle to move by, in degrees (positive -> forward, negative -> backward)
        :param blocking: if True, wait for the motor to complete the move before continuing execution
        :param timeout: how many seconds to wait for move to complete
        :raises HardwareException: if timeout exceeded
        :returns: boolean indicating success
        """

        if self.in_motion:
            self.stop()

        puts(f'Moving motor by {amt}...')
        scaled = amt/360

        self.stage.move_completed = False
        self.stage.moveby(scaled)

        if blocking:
            t = 0
            while not self.stage.move_completed:
                time.sleep(0.1)
                t += 0.1
                if t >= timeout:
                    puterr(f'Exceeded time limit waiting for motor to move by {amt}')
                    raise HardwareException(str(self.serial_number) + ' moveby')
                    return False

            puts('Moved motor')
        return True

[docs]    def jog(self, d:int=1):
        """
        Move the motor continuously in a given direction (non-blocking)

        Example::

            import time

            motor = DDR25(12345678)

            motor.jog_max_velocity = 100 # set jog max velocity to 100 deg/sec
            motor.jog_acceleration = 200 # set jog acceleration to 200 deg/sec^2

            motor.jog(1) # jog forward
            time.sleep(5)
            motor.jog(2) # jog backward
            time.sleep(5)

            motor.stop()
        
        :param d: direction (1 for forward, 2 for backward)
        :returns: None
        """

        if d not in (1, 2):
            puterr(f'Given invalid direction {d}. Must be 1 or 2')
            return
        if self.in_motion:
            self.stop()
        puts('Jogging motor ' + ('forward' if d == 1 else 'backward') + '...')
        self.stage._port.send_message(MGMSG_MOT_MOVE_JOG(chan_ident = self.stage._chan_ident, direction=d))



[docs]    def stop(self, immediately:bool=False, blocking:bool=True, timeout:float=5) -> bool:
        """
        Stop the motor

        Does nothing and returns ``False`` if motor already stopped
        
        Example::

            import time
            
            motor = DDR25(12345678)
            motor.home()
            motor.max_velocity = 10 # 10 deg/sec
            motor.acceleration = 100 # 100 deg/sec^2

            motor.goto(100, blocking=False) # non-blocking goto
            time.sleep(3) # wait 3 seconds

            motor.stop() # stop the motor
            print(motor.position) # should be somewhere between 0 and 100

            # since jogs are non-blocking and have no destination, the only way to stop a jog is stop()
            motor.jog(1)
            motor.stop()

        :param immediately: if True, stop the motor immediately. Otherwise, stop gradually (profiled)
        :param blocking: if True, wait for the motor to finish stopping before continuing execution
        :param timeout: how many seconds to wait for stop to complete
        :returns: boolean indicating success
        """

        if not self.in_motion:
            putwarning('Motor already stopped')
            return False

        puts('Stopping motor...')
        self.stage.move_stopped = False
        mode = 1 if immediately else 2
        self.stage._port.send_message(MGMSG_MOT_MOVE_STOP(chan_ident = self.stage._chan_ident, stop_mode=mode))

        if blocking:
            t = 0
            while not self.stage.move_stopped:
                time.sleep(0.1)
                t += 0.1
                if t >= timeout:
                    puterr('Exceeded timeout waiting for motor to stop')
                    raise HardwareException(str(self.serial_number) + ' stop')
                    return False
            puts('Stopped motor')
        return True

    @property

        (Read-only) Whether or not the motor is in motion

        "In motion" includes moving (goto/moveby), jogging, and homing
        """
        return self.stage.status_in_motion_forward or self.stage.status_in_motion_reverse or self.stage.status_in_motion_jogging_forward or self.stage.status_in_motion_jogging_reverse or self.stage.status_in_motion_homing

    @property
    def is_homing(self) -> bool:
        """(Read-only) Whether or not the motor is currently homing"""
        return self.stage.status_in_motion_homing

    @property
    def is_jogging(self) -> bool:
        """(Read-only) Whether or not the motor is currently jogging"""
        return self.stage.status_in_motion_jogging_forward or self.stage.status_in_motion_jogging_backward

    @property
    def position(self) -> float:
        """
        (Read-only) The current position of the motor, in degrees (float in ``[0,360)``)

        Example::

            import time

            motor = DDR25(12345678)

            motor.home()
            print(motor.position) # should be 0

            motor.goto(45.6)
            print(motor.position) # should be close to 45.6

            motor.moveby(10)
            print(motor.position) # should be close to 55.6
            
            motor.jog_max_velocity = 10
            motor.jog() # jog forward
            
            # should print an ascending sequence of angles
            for _ in range(10):
                print(motor.position)
                time.sleep(1)

            motor.stop()
            print(motor.position)
        """
        pos = self.stage.position
        if pos < 0:
            pos += 1
        pos *= 360
        return pos % 360

    # the current velocity of the stage (NOT the maximum velocity)
    @property
    def velocity(self) -> float:
        """
        (Read-only) The current velocity of the stage, in degrees per second

        - Positive means rotating forward (angle increasing)
        - Negative means rotating backward (angle decreasing)
        - Zero means stationary

        Do not confuse with :attr:`.DDR25.max_velocity`, which is the maximum velocity that goto() and moveby() will make the motor move at.

        Example::

            import time

            motor = DDR25(12345678)
            motor.home()
            motor.max_velocity = 30
            motor.acceleration = 5

            # should print an increasing sequence of velocities before plateauing at 30
            motor.moveby(10000, blocking=False) # move forward for a very long time
            for _ in range(10):
                print(motor.velocity)
                time.sleep(1)

            # should print a decreasing sequence of velocities before plateauing at -30
            motor.moveby(-10000, blocking=False) # move backward for a very long time
            for _ in range(10):
                print(motor.velocity)
                time.sleep(1)

            motor.stop()
        """
        vel = self.stage.velocity
        vel *= 360
        return vel

    @property
    def min_velocity(self) -> float:
        """
        (Read-and-write) Minimum velocity of moveby()/goto() movements, in degrees per second

        Typically 0
        
        Although this property is read-and-write,
        writing to it doesn't do anything.
        """

        vel = self.stage.min_velocity
        vel *= 360
        return vel

    # the velocity which the stage will move at when jogging, for example
    @property
    def max_velocity(self) -> float:
        """
       (Read-and-write)  Maximum velocity of moveby()/goto() movements, in degrees per second

        When moveby() or goto() is called, the motor will not exceed the set
        maximum velocity while moving toward its destination.
        
        Do not confuse with :attr:`.DDR25.velocity`, which is the current physical velocity of the motor
        """
        vel = self.stage.max_velocity
        vel *= 360
        return vel

    @property
    def acceleration(self) -> float:
        """
       (Read-and-write) Acceleraiton of moveby()/goto() movements, in degrees per second^2
        
        When moveby() or goto() is called, the motor will accelerate at the
        given rate until it reaches the maximum velocity.
        """
        acc = self.stage.acceleration
        acc *= 360
        return acc
