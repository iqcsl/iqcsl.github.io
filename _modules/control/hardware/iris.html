    """
    Thorlabs ELL15 Iris class

    Typically connected to an Interface Board, which is connected to a computer

    Example::
        
        from control.hardware.iris import ELL15

        iris = ELL15('DK0BJ258') # replace with interface board serial number
        print(iris.info())
        print(iris.status())
        print(iris.max_aperture)
        print(iris.min_aperture)
        iris.home()
        iris.goto(5)
        iris.moverel(-0.5)
        print(iris.position)
        
    """
    def __init__(self, sn):
        """
        :type sn: int or str
        :param sn: serial number (of interface board, not iris)

        :raises Exception: if fails to connect
        """
        self.serial_number = sn
        """Serial number"""

        devices = list_devices()
        dev = None
        for d in devices:
            if d['serial_number'] == str(sn):
                dev = d
                break
        if dev == None:
            puterr('Could not locate ELL15 device with serial number ' + str(sn))
            raise Exception

        port_name = dev['device']
        self._port = serial.Serial(port=port_name,
                                   baudrate=9600,
                                   parity=serial.PARITY_NONE,
                                   stopbits=serial.STOPBITS_ONE,
                                   bytesize=serial.EIGHTBITS,
                                   timeout=10.0,
                                   write_timeout=1.0)

        self._port.reset_input_buffer()
        self._port.reset_output_buffer()

        inf = self.info()
        self._pulse_per_mm = inf['pulse'] # encoder pulses per millimeter
        self._min_pulse = 1000 # this is not obtainable using info(), so we use the ELL15 value given by thorlabs
        self._max_pulse = inf['travel']+self._min_pulse


    def __del__(self):
        try:
            self._port.close()
            print('Closed iris serial port')
        except:
            pass

    @property
    def is_connected(self) -> bool:
        """
        Whether the device is connected
        
        Tries to call :meth:`.ELL15.status()` and returns True or False based on whether it succeeds
        """
        try:
            self.status()
            return True
        except Exception as e:
            log('ELL15.is_connected() encountered the following exception:')
            log(repr(e))
            log(traceback.format_exc())
            return False


    def _write(self, msg):
        self._port.write(bytearray(msg, 'ascii'))

    # read until \r\n and return what was read
    def _read(self, timeout=5):
        t = 0
        resp = b''
        while t < timeout:
            w = self._port.inWaiting()
            for _ in range(w):
                resp += self._port.read(1)

                # end of message '\r\n'
                if len(resp) >= 2 and resp[-2] == ord('\r') and resp[-1] == ord('\n'):
                    s = resp.decode('ascii')
                    return s[:-2]

            time.sleep(0.01)
            t += 0.01

        # timeout
        puterr(f'Timeout while waiting for response from ELL15 {self.serial_number}')
        raise HardwareException(str(self.serial_number) + ' _read()')
        return ''

[docs]    def info(self, addr:int=0):
        """
        Get info about the device

        Includes:

        - Default address (see manual)
        - ELL model number
        - Serial number
        - Year of manufacture
        - Firmware release
        - Hardware release (see manual)
        - Travel (mm)
        - Encoder pulses per position

        :param addr: address
        :returns: a ``str -> Any`` dictionary containing device information
        """
        puts('Asking iris for info...')
        self._port.reset_input_buffer()
        self._write(str(addr) + 'in') # e.x. '0in'
        resp = self._read()

        address = int(resp[0],16)
        assert(resp[1:3] == 'IN')
        ell_nbr = int(resp[3:5],16)
        serial_nbr = resp[5:13]
        year = resp[13:17]
        firm = int(resp[17:19])

        temp = int(resp[19:21],16)
        if temp >= 0b10000000:
            unit = 'imperial'
            temp -= 0b10000000
        else:
            unit = 'metric'
        hard = temp

        travel = int(resp[21:25],16)
        pulse = int(resp[25:33],16)

        return {'address':address, 'ell_number':ell_nbr, 'serial_number':serial_nbr, 'year':year, 'firmware':firm, 'unit':unit, 'hardware':hard, 'travel':travel, 'pulse':pulse}



[docs]    def status(self, addr:int=0) -> str:
        """
        Get status of device

        Possible status strings are:

        - OK
        - Communication time out
        - Mechanical time out
        - Command error or not supported
        - Value out of range
        - Module isolated
        - Module out of isolation
        - Initializing error
        - Thermal error
        - Busy
        - Sensor error
        - Motor error
        - Out of Range
        - Over Current error
        - Unknown error

        See the manual for more details
        
        :param addr: address
        :returns: status string
        """
        puts('Asking iris for status...')
        self._port.reset_input_buffer()
        self._write(str(addr) + 'gs')
        resp = self._read()
        assert resp[1:3] == 'GS'

        return STATUS_STR[int(resp[3:5],16)]


    # many commands have the motor either return GS or PO
    # this function processes that output
    def _process_pos(self,resp):
        if resp[1:3] == 'GS':
            stat = int(resp[3:5],16)
            log('Status: ' + STATUS_STR[stat])
            return -1
        else:
            assert resp[1:3] == 'PO'
            pulse = int(resp[3:11], 16)
            return pulse/self._pulse_per_mm


[docs]    def home(self, addr:int=0, d:int=0, blocking:bool=True):
        """
        Home the iris

        For the ELL15, home is the maximum aperture by default.

        :param addr: address
        :param d: homing direction (for rotary stages only)
        :param blocking: if True, wait for homing to finish stopping before continuing execution
        :returns: boolean indicating success
        """
        self._port.reset_input_buffer()
        self._write(str(addr) + 'ho' + str(d))

        if blocking:
            resp = self._read()
            # return self._process_pos(resp)
            puts('Homed iris')
        return True

[docs]    def goto(self, mm:float, addr:int=0, blocking:bool=True):
        """
        Go to a certain position

        The given position must be in the interval ``[ELL15.min_aperture, ELL15.max_aperture]``.

        :param mm: target position in millimeters
        :param addr: address
        :param blocking: if True, wait for move to finish stopping before continuing execution
        """
        #print('setting iris to ' + str(mm))
        pulses = int(mm * self._pulse_per_mm)

        if pulses < self._min_pulse or pulses > self._max_pulse:
            puterr(f'ELL15.goto(): Out of bounds ({mm}mm)')
            return -1

        # 8 character long hex string (no leading 0x)
        hexstr = hex(pulses)[2:].zfill(8)
        self._port.reset_input_buffer()
        self._write(str(addr) + 'ma' + hexstr)
        if blocking:
            # response is either a PO (position) or GS (get status)
            # return -1 if there's a problem,
            # otherwise return current position in encoder pulses
            resp = self._read()
            # return self._process_pos(resp)
            puts(f'Moved iris to {mm}')
        return True

[docs]    def moverel(self, mm:float, addr:int=0, blocking:bool=True):
        """
        Move relatively by a certain amount
        
        The given amount must not move the iris out of the interval ``[ELL.min_aperture, ELL15.max_aperture]``.
        
        :param mm: amount to move in millimeters. Positive means increase aperture, negative means decrease aperture
        :param addr: address
        :param blocking: if True, wait for move to finish stopping before continuing execution
        """
        # destination, in millimeters
        dest = self.position + mm
        pulses = int(dest * self._pulse_per_mm)
        if pulses < self._min_pulse or pulses > self._max_pulse:
            puterr(f'ELL15.moverel(): Out of bounds ({dest}mm)')
            return False

        hexstr = hex(pulses)[2:].zfill(8)
        self._port.reset_input_buffer()
        self._write(str(addr) + 'ma' + hexstr)
        if blocking:
            resp = self._read()
            # return self._process_pos(resp)
            puts(f'Moved iris to {dest}')
        return True


    @property
    def position(self, addr:int=0) -> float:
        """
        Current position
        
        :param addr: address
        :returns: current aperture size in millimeters
        """
        self._port.reset_input_buffer()
        self._write(str(addr) + 'gp')
        resp = self._read()
        assert resp[1:3] == 'PO'

        pulse = int(resp[3:11],16)
        return pulse/self._pulse_per_mm

    @property
    def max_aperture(self) -> float:
        """
        Maximum aperture size
        
        :returns: maximum aperture size in millimeters
        """
        return self._max_pulse / self._pulse_per_mm

    @property
    def min_aperture(self) -> float:
        """
        Minimum aperture size

        :returns: minimum aperture size in millimeters
        """
        return self._min_pulse / self._pulse_per_mm
