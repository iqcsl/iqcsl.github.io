[docs]def initMotor(sn) -> kin.KinesisMotor:
    """
    Search for motor with given serial number and connect to it

    :type sn: str or int
    :param sn: serial number
    :returns: pyLabLib KinesisMotor object
    """
    mot = kin.KinesisMotor(str(sn))
    return mot


# in pylablib, if the motor is disabled and you send it a movement command, the program will hang

[docs]def enable_first(fn):
    def inner(*args, **kwargs):
        if not args[0].is_enabled:
            args[0].enable()
        return fn(*args, **kwargs)
    return inner



[docs]class DDR25:
    encs_per_rev = 1440000

    @classmethod
    def enc2deg(cls, enc):
        return enc / cls.encs_per_rev * 360
    @classmethod
    def deg2enc(cls, deg):
        return deg / 360 * cls.encs_per_rev

    def __init__(self, sn):
        self.stage = initMotor(sn)
        self.serial_number = sn

    def __del__(self):
        try:
            self.close()
        except:
            pass
    
    @property
    def is_connected(self):
        try:
            self.identify()
            return True
        except Exception as e:
            log('DDR25.is_connected() encountered the following exception (returning False):')
            log(repr(e))
            log(traceback.format_exc())
            return False

    def close(self):
        self.stage.close()
        log(f'Disconnected {self.__class__.__name__} with serial number {self.serial_number}')

    def identify(self):
        puts('Identifying motor...')
        self.stage.blink()

    @enable_first
    def home(self, blocking=True, timeout=5):
        puts('Homing motor...')
        if self.in_motion:
            self.stop()
        self.stage.home(sync=blocking, force=True, timeout=timeout)

    @enable_first
    def goto(self, pos, blocking=True, timeout=5):
        if self.in_motion:
            self.stop()
        # mod the result so goto(1000000) doesn't move forever
        encs = self.deg2enc(pos) % self.encs_per_rev

        puts(f'Moving motor to {pos}...')
        self.stage.move_to(encs)

        if blocking:
            try:
                self.stage.wait_move(timeout=timeout)
            except ThorlabsTimeoutError:
                puterr(f'Exceeded time limit waiting for motor to goto position {pos}')
            else:
                puts(f'Moved motor to {pos}')

    @enable_first
    def moveby(self, amt, blocking=True, timeout=5):
        if self.in_motion:
            self.stop()
        puts(f'Moving motor by {amt}...')

        encs = self.deg2enc(amt)
        try:
            self.stage.move_by(encs)
        except struct.error:
            puterr(f'moveby(): out of bounds {amt}')
            return

        if blocking:
            try:
                self.stage.wait_move(timeout=timeout)
            except ThorlabsTimeoutError:
                puterr(f'Exceeded time limit waiting for motor to move by {amt}')
            else:
                puts(f'Moved motor by {amt}')

    # d=1: forward
    # d=2: backward
    @enable_first
    def jog(self, d=1, kind='builtin'):
        if d not in (1, 2):
            puterr(f'Given invalid direction {d}. Must be 1 or 2')
            return
        if self.in_motion:
            self.stop()
        puts('Jogging motor ' + ('forward' if d == 1 else 'backward') + '...')
        
        self.stage.jog('+' if d==1 else '-', kind=kind)

    @enable_first
    def stop(self, immediately=False, blocking=True, timeout=5):
        puts('Stopping motor...')

        self.stage.stop(immediate=immediately, sync=blocking, timeout=timeout)

    def enable(self):
        self.stage._enable_channel()

    def disable(self):
        self.stage._enable_channel(False)

    @property
    def is_enabled(self):
        return 'enabled' in self.stage.get_status()

    @property

    @property

    def get_velocity_parameters(self):
        vp = self.stage.get_velocity_parameters()
        # scale everything
        scaled_vp = tuple(self.enc2deg(x) for x in vp)
        return scaled_vp

    def set_velocity_parameters(self, minvel, accel, maxvel):
        self.stage.setup_velocity(
                min_velocity = self.deg2enc(minvel),
                acceleration = self.deg2enc(accel),
                max_velocity = self.deg2enc(maxvel))

    def set_jog_parameters(self, mode=None, step=None, minvel=None, accel=None, maxvel=None, stopmode=None):
        self.stage.setup_jog(
                mode = mode,
                step_size = step,
                min_velocity = self.deg2enc(minvel) if minvel is not None else None,
                acceleration = self.deg2enc(accel) if accel is not None else None,
                max_velocity = self.deg2enc(maxvel) if maxvel is not None else None,
                stop_mode = stopmode)

    def get_jog_parameters(self):
        jp = self.stage.get_jog_parameters()
        scaled_jp = (
                jp[0],
                jp[1],
                self.enc2deg(jp[2]),
                self.enc2deg(jp[3]),
                self.enc2deg(jp[4]),
                jp[5])

        return scaled_jp
