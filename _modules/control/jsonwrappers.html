import json

# returns a dict loaded from given JSON file name
# if anything goes wrong, returns empty dict

[docs]def filename2dict(fname):
    try:
        with open(fname, 'r') as f:
            return str2dict(f.read())
    except Exception as e:
        print(repr(e))
        return {}



[docs]def str2dict(s):
    data = json.loads(s)
    if not isinstance(data, dict):
        print(f'given JSON data that is not a dictionary')
        return {}
    return data
